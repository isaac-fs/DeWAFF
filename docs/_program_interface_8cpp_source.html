255 Mat ProgramInterface::processFrame(const Mat &inputFrame) {
256  // Process frame
257  Mat input = inputPreProcessor(inputFrame);
258  Mat output;
259  switch (filterType) {
260  case DBF:
261  output = framework.DeceivedBilateralFilter(input, windowSize, spatialSigma, rangeSigma);
262  break;
263  case DSBF:
264  output = framework.DeceivedScaledBilateralFilter(input, windowSize, spatialSigma, rangeSigma);
265  break;
266  case DNLMF:
267  output = framework.DeceivedNonLocalMeansFilter(input, windowSize, neighborhoodSize, spatialSigma, rangeSigma);
268  break;
269  case DGF:
270  output = framework.DeceivedGuidedFilter(input, windowSize, spatialSigma, rangeSigma);
271  break;
272  default:
273  help();
274  break;
275  }
276 
277  return outputPosProcessor(output);

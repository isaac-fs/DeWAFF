93 cv::Mat GuidedFilterMono::filterSingleChannel(const cv::Mat &p) const {
94  cv::Mat mean_p = boxfilter(p, r);
95  cv::Mat mean_Ip = boxfilter(I.mul(p), r);
96  cv::Mat cov_Ip = mean_Ip - mean_I.mul(mean_p); // this is the covariance of (I, p) in each local patch.
97 
98  cv::Mat a = cov_Ip / (var_I + eps); // Eqn. (5) in the paper;
99  cv::Mat b = mean_p - a.mul(mean_I); // Eqn. (6) in the paper;
100 
101  cv::Mat mean_a = boxfilter(a, r);
102  cv::Mat mean_b = boxfilter(b, r);
103 
104  return mean_a.mul(I) + mean_b;

107 GuidedFilterColor::GuidedFilterColor(const cv::Mat &origI, int r, double eps) : r(r), eps(eps) {
108  cv::Mat I;
109  if (origI.depth() == CV_32F || origI.depth() == CV_64F)
110  I = origI.clone();
111  else
112  I = convertTo(origI, CV_32F);
113 
114  Idepth = I.depth();
115 
116  cv::split(I, Ichannels);
117 
118  mean_I_r = boxfilter(Ichannels[0], r);
119  mean_I_g = boxfilter(Ichannels[1], r);
120  mean_I_b = boxfilter(Ichannels[2], r);
121 
122  // variance of I in each local patch: the matrix Sigma in Eqn (14).
123  // Note the variance in each local patch is a 3x3 symmetric matrix:
124  // rr, rg, rb
125  // Sigma = rg, gg, gb
126  // rb, gb, bb
127  cv::Mat var_I_rr = boxfilter(Ichannels[0].mul(Ichannels[0]), r) - mean_I_r.mul(mean_I_r) + eps;
128  cv::Mat var_I_rg = boxfilter(Ichannels[0].mul(Ichannels[1]), r) - mean_I_r.mul(mean_I_g);
129  cv::Mat var_I_rb = boxfilter(Ichannels[0].mul(Ichannels[2]), r) - mean_I_r.mul(mean_I_b);
130  cv::Mat var_I_gg = boxfilter(Ichannels[1].mul(Ichannels[1]), r) - mean_I_g.mul(mean_I_g) + eps;
131  cv::Mat var_I_gb = boxfilter(Ichannels[1].mul(Ichannels[2]), r) - mean_I_g.mul(mean_I_b);
132  cv::Mat var_I_bb = boxfilter(Ichannels[2].mul(Ichannels[2]), r) - mean_I_b.mul(mean_I_b) + eps;
133 
134  // Inverse of Sigma + eps * I
135  invrr = var_I_gg.mul(var_I_bb) - var_I_gb.mul(var_I_gb);
136  invrg = var_I_gb.mul(var_I_rb) - var_I_rg.mul(var_I_bb);
137  invrb = var_I_rg.mul(var_I_gb) - var_I_gg.mul(var_I_rb);
138  invgg = var_I_rr.mul(var_I_bb) - var_I_rb.mul(var_I_rb);
139  invgb = var_I_rb.mul(var_I_rg) - var_I_rr.mul(var_I_gb);
140  invbb = var_I_rr.mul(var_I_gg) - var_I_rg.mul(var_I_rg);
141 
142  cv::Mat covDet = invrr.mul(var_I_rr) + invrg.mul(var_I_rg) + invrb.mul(var_I_rb);
143 
144  invrr /= covDet;
145  invrg /= covDet;
146  invrb /= covDet;
147  invgg /= covDet;
148  invgb /= covDet;
149  invbb /= covDet;
150 }

152 cv::Mat GuidedFilterColor::filterSingleChannel(const cv::Mat &p) const {
153  cv::Mat mean_p = boxfilter(p, r);
154 
155  cv::Mat mean_Ip_r = boxfilter(Ichannels[0].mul(p), r);
156  cv::Mat mean_Ip_g = boxfilter(Ichannels[1].mul(p), r);
157  cv::Mat mean_Ip_b = boxfilter(Ichannels[2].mul(p), r);
158 
159  // covariance of (I, p) in each local patch.
160  cv::Mat cov_Ip_r = mean_Ip_r - mean_I_r.mul(mean_p);
161  cv::Mat cov_Ip_g = mean_Ip_g - mean_I_g.mul(mean_p);
162  cv::Mat cov_Ip_b = mean_Ip_b - mean_I_b.mul(mean_p);
163 
164  cv::Mat a_r = invrr.mul(cov_Ip_r) + invrg.mul(cov_Ip_g) + invrb.mul(cov_Ip_b);
165  cv::Mat a_g = invrg.mul(cov_Ip_r) + invgg.mul(cov_Ip_g) + invgb.mul(cov_Ip_b);
166  cv::Mat a_b = invrb.mul(cov_Ip_r) + invgb.mul(cov_Ip_g) + invbb.mul(cov_Ip_b);
167 
168  cv::Mat b = mean_p - a_r.mul(mean_I_r) - a_g.mul(mean_I_g) - a_b.mul(mean_I_b); // Eqn. (15) in the paper;
169 
170  return (boxfilter(a_r, r).mul(Ichannels[0])
171  + boxfilter(a_g, r).mul(Ichannels[1])
172  + boxfilter(a_b, r).mul(Ichannels[2])
173  + boxfilter(b, r)); // Eqn. (16) in the paper;
174 }

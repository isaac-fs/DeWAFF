99 Mat Utils::LoGFilter(const Mat &image, int windowSize, double sigma) {
100  // Get the Gaussian kernel
101  Mat gaussianKernel = GaussianKernel(windowSize, sigma);
102 
103  // Pre computation of meshgrid values
104  Mat X, Y, S;
105  Range range = Range(-(windowSize/2), (windowSize/2) + 1);
106  MeshGrid(range, X, Y);
107  pow(X, 2.0, X);
108  pow(Y, 2.0, Y);
109 
110  // Variance
111  double variance = pow(sigma, 2.0);
112  Mat laplacianOfGaussianKernel = (1.0 / (2.0 * CV_PI * variance)) * (((X+Y)/variance) - 2.0).mul(gaussianKernel);
113 
114  // Normalization
115  laplacianOfGaussianKernel -= sum(laplacianOfGaussianKernel).val[0] / pow(windowSize, 2.0);
116

118  Mat LoGFilteredImage(image.size(), image.type());
119 
120  // Apply the Laplacian filter
121  filter2D(image, LoGFilteredImage, -1, laplacianOfGaussianKernel, Point(-1,-1), 0, BORDER_CONSTANT);

160  Mat inputImage;

164

168  yRange = Range(windowSize/2, windowSize/2 + neighborhoodSize);

171  Mat fixedWindow = inputImage(xRange, yRange);
172 
173  // Moving pixel sub region
174  Mat slidingWindow(fixedWindow.size(), fixedWindow.type());
175 
176  // Initialize the output matrix
177  Mat distancesMatrix(inputImage_.size(), inputImage_.type());
178 
179  // Visit each pixel in the image region
180  for(int i = 0; i < windowSize; i++) {
181  xRange = Range(i, i + neighborhoodSize);

183  yRange = Range(j, j + neighborhoodSize);

186  slidingWindow = inputImage(xRange, yRange);
187 
188  // Calculate the euclidean distance between patches
